### enviar_comando_roku()

Sends HTTP commands to Roku device.

<ParamField path="ip" type="string" required>
  IP address of the Roku device
</ParamField>

<ParamField path="endpoint" type="string" required>
  API endpoint (e.g., "keypress/Home", "launch/12")
</ParamField>

<ParamField path="metodo" type="string" default="POST">
  HTTP method - "POST" or "GET"
</ParamField>

<ResponseField name="success" type="bool">
  Returns `True` if command succeeded, `False` otherwise
</ResponseField>

<CodeGroup>
```python Basic Commands
import requests

def enviar_comando_roku(ip, endpoint, metodo="POST"):
    """Send command to Roku device."""
    url = f"http://{ip}:8060/{endpoint}"
    try:
        if metodo == "POST":
            requests.post(url, timeout=5)
        else:
            requests.get(url, timeout=5)
        return True
    except:
        return False

# Examples
ip = "192.168.1.8"

# Launch Netflix
enviar_comando_roku(ip, "launch/12")

# Press Home button
enviar_comando_roku(ip, "keypress/Home")

# Power off
enviar_comando_roku(ip, "keypress/PowerOff")

# Volume up
enviar_comando_roku(ip, "keypress/VolumeUp")
```

### obtener_app_actual()

Retrieves the currently active application on Roku device.

<ParamField path="ip" type="string" required>
  IP address of the Roku device
</ParamField>

<ResponseField name="app_name" type="string">
  Name of the active application, or "el Menú de Inicio" if on home screen, or "desconocida" if unable to determine
</ResponseField>

<CodeGroup>
```python Usage
def obtener_app_actual(ip):
    """Get currently active app on Roku."""
    url = f"http://{ip}:8060/query/active-app"
    try:
        response = requests.get(url, timeout=1.5)
        if response.status_code == 200:
            match = re.search(r'<app.*?>(.*?)</app>', response.text)
            if match:
                app_name = match.group(1)
                return app_name if app_name != "Roku" else "el Menú de Inicio"
    except:
        pass
    return "desconocida"

# Example

app = obtener_app_actual(ip)

```

### ejecutar_busqueda_roku()

<ParamField path="ip" type="string" required>
  IP address of the Roku device
</ParamField>

<ParamField path="app_name" type="string" required>
  Name of the application to search in (e.g., "netflix", "youtube")
</ParamField>

<ParamField path="query" type="string" required>
  Search query text
</ParamField>

<ResponseField name="success" type="bool">
  Returns `True` if search was executed successfully
</ResponseField>

<CodeGroup>
```python Implementation
import requests
import time

def ejecutar_busqueda_roku(ip, app_name, query):
    """Execute search in Roku app."""
    # Get app ID from mapping
    app_id = ROKU_APPS.get(app_name.lower(), "837")  # Default to YouTube
    
    if app_id:
        # Launch the app
        enviar_comando_roku(ip, f"launch/{app_id}")
        time.sleep(2)  # Wait for app to load
        
        # URL encode the query
        query_encoded = requests.utils.quote(query)
        
        # Execute search
        endpoint = f"search/browse?keyword={query_encoded}&launch=true&provider-id={app_id}"
        return enviar_comando_roku(ip, endpoint)
    
    return False

# Examples
ip = "192.168.1.8"

# Search YouTube
ejecutar_busqueda_roku(ip, "youtube", "recetas de cocina")

# Search Netflix
ejecutar_busqueda_roku(ip, "netflix", "películas de acción")

# Search with special characters
ejecutar_busqueda_roku(ip, "prime", "El Señor de los Anillos")
```

### verificar_roku_por_info()

Verifies if a device at given IP is a Roku device.

<ParamField path="ip" type="string" required>
  IP address to check
</ParamField>

<ResponseField name="device_info" type="dict|None">
  Returns device information dict with 'ip' and 'tipo' keys, or `None` if not a Roku device
</ResponseField>

<CodeGroup>
```python Implementation
import requests
import re

def verificar_roku_por_info(ip):
    """Check if device is a Roku."""
    url = f"http://{ip}:8060/query/device-info"
    try:
        response = requests.get(url, timeout=1.5)
        if response.status_code == 200 and "<device-info>" in response.text:
            # Extract model name
            model_match = re.search(
                r'<model-name>(.*?)</model-name>',
                response.text
            )
            model_name = model_match.group(1) if model_match else "Roku"
            
            return {
                "ip": ip,
                "tipo": f"Roku ({model_name})"
            }
    except:
        pass
    return None

# Usage
device = verificar_roku_por_info("192.168.1.8")
if device:
    print(f"Found: {device['tipo']} at {device['ip']}")
else:
    print("No Roku device found")
```

```python Parse Device Info
def get_full_device_info(ip):
    """Get complete device information."""
    url = f"http://{ip}:8060/query/device-info"
    response = requests.get(url, timeout=1.5)
    
    if response.status_code == 200:
        text = response.text
        
        # Extract various fields
        info = {}
        patterns = {
            'model_name': r'<model-name>(.*?)</model-name>',
            'model_number': r'<model-number>(.*?)</model-number>',
            'serial_number': r'<serial-number>(.*?)</serial-number>',
            'software_version': r'<software-version>(.*?)</software-version>',
            'device_name': r'<user-device-name>(.*?)</user-device-name>',
        }
        
        for key, pattern in patterns.items():
            match = re.search(pattern, text)
            if match:
                info[key] = match.group(1)
        
        return info
    
    return None

# Example
info = get_full_device_info("192.168.1.8")
for key, value in info.items():
    print(f"{key}: {value}")
```

### escanear_red_tvs()

Scans local network for Roku devices.

<ResponseField name="dispositivos" type="list[dict]">
  List of discovered devices with 'ip' and 'tipo' keys
</ResponseField>

<CodeGroup>
```python Implementation
import socket
import threading
import time

def escanear_red_tvs(target_ip="192.168.1.8"):
    """Scan network for Roku devices."""
    encontrados = []
    
    # Try target IP first
    tv_principal = verificar_roku_por_info(target_ip)
    if tv_principal:
        encontrados.append(tv_principal)
        return encontrados
    
    try:
        # Get local IP to determine network range
        s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        s.connect(("8.8.8.8", 80))
        ip_local = s.getsockname()[0]
        s.close()
        
        # Extract network prefix (e.g., "192.168.1")
        prefijo = ".".join(ip_local.split(".")[:-1])
        
        # Scan function for threading
        def tarea_escaneo(ip_test):
            res = verificar_roku_por_info(ip_test)
            if res:
                encontrados.append(res)
        
        # Scan all IPs in range
        hilos = []
        for i in range(1, 255):
            t = threading.Thread(
                target=tarea_escaneo,
                args=(f"{prefijo}.{i}",)
            )
            t.daemon = True
            t.start()
            hilos.append(t)
        
        # Wait for scan to complete
        time.sleep(1.5)
        
    except:
        pass
    
    return encontrados

# Usage
devices = escanear_red_tvs()
print(f"Found {len(devices)} Roku device(s):")
for device in devices:
    print(f"  {device['tipo']} at {device['ip']}")
```

```python Optimized Scan
import concurrent.futures

def scan_network_fast(timeout=2.0):
    """Fast network scan using ThreadPoolExecutor."""
    # Get network prefix
    s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    s.connect(("8.8.8.8", 80))
    ip_local = s.getsockname()[0]
    s.close()
    prefijo = ".".join(ip_local.split(".")[:-1])
    
    # Generate all IPs to scan
    ips_to_scan = [f"{prefijo}.{i}" for i in range(1, 255)]
    
    # Scan with thread pool
    devices = []
    with concurrent.futures.ThreadPoolExecutor(max_workers=50) as executor:
        future_to_ip = {
            executor.submit(verificar_roku_por_info, ip): ip
            for ip in ips_to_scan
        }
        
        for future in concurrent.futures.as_completed(
            future_to_ip,
            timeout=timeout
        ):
            result = future.result()
            if result:
                devices.append(result)
    
    return devices

# Usage
devices = scan_network_fast(timeout=3.0)
print(f"Scan complete: {len(devices)} device(s) found")
```

### hablar_local()

Generates speech from text using Google Text-to-Speech.

<ParamField path="texto" type="string" required>
  Text to convert to speech (automatically cleaned)
</ParamField>

<CodeGroup>
```python Implementation
import os
import time
import re
from gtts import gTTS
import pygame

def limpiar_texto_para_audio(texto):
    """Clean text for TTS."""
    # Remove command tags
    texto = re.sub(r'/\*.*?\*/', '', texto)
    # Remove markdown
    texto = texto.replace('*', '').replace('#', '').replace('_', ' ')
    # Remove URLs
    texto = re.sub(r'http\S+', '', texto)
    # Normalize whitespace
    texto = " ".join(texto.split())
    return texto

def hablar_local(texto):
    """Speak text using gTTS."""
    texto_limpio = limpiar_texto_para_audio(texto)
    if not texto_limpio:
        return
    
    try:
        # Generate speech
        tts = gTTS(text=texto_limpio, lang='es', slow=False)
        filename = "temp_voice.mp3"
        tts.save(filename)
        
        # Play audio
        pygame.mixer.init()
        pygame.mixer.music.load(filename)
        pygame.mixer.music.play()
        
        # Wait for playback to finish
        while pygame.mixer.music.get_busy():
            time.sleep(0.1)
        
        # Cleanup
        pygame.mixer.music.unload()
        if os.path.exists(filename):
            os.remove(filename)
            
    except Exception as e:
        print(f"Audio error: {e}")

# Usage
hablar_local("Hola Rosario, ¿cómo estás hoy?")
```

```python Parse Commands
import re

def parse_commands(text):
    """Parse command tags into structured format."""
    commands = []
    tags = re.findall(r'/\*(.*?)\*/', text)
    
    for tag in tags:
        # app(name)
        if 'app(' in tag:
            match = re.search(r'app\((.*?)\)', tag)
            if match:
                commands.append({
                    'type': 'app',
                    'value': match.group(1)
                })
        
        # search(app, query)
        elif 'search(' in tag:
            match = re.search(r'search\((.*?),\s*(.*?)\)', tag)
            if match:
                commands.append({
                    'type': 'search',
                    'app': match.group(1),
                    'query': match.group(2)
                })
        
        # resultados(query)
        elif 'resultados(' in tag:
            match = re.search(r'resultados\((.*?)\)', tag)
            if match:
                commands.append({
                    'type': 'search_web',
                    'query': match.group(1)
                })
        
        # Simple commands
        elif tag in ['home', 'power', 'salir']:
            commands.append({'type': tag})
    
    return commands

# Example
text = "/*app(netflix)*/ Abriendo Netflix. /*search(youtube, gatos)*/"
commands = parse_commands(text)
for cmd in commands:
    print(cmd)
# {'type': 'app', 'value': 'netflix'}
# {'type': 'search', 'app': 'youtube', 'query': 'gatos'}
```
</CodeGroup>

## Memory Management

### Conversation History

Manage conversation context for AI models.

<CodeGroup>
```python Basic History
class ConversationHistory:
    def __init__(self, max_length=10):
        self.history = []
        self.max_length = max_length
    
    def add(self, user_input, ai_response):
        """Add conversation turn."""
        self.history.append({
            'u': user_input,
            'k': ai_response
        })
        
        # Trim if too long
        if len(self.history) > self.max_length:
            self.history.pop(0)
    
    def get_formatted(self):
        """Get formatted history for API."""
        formatted = []
        for turn in self.history:
            formatted.append({
                'role': 'user',
                'content': turn['u']
            })
            formatted.append({
                'role': 'assistant',
                'content': turn['k']
            })
        return formatted
    
    def clear(self):
        """Clear history."""
        self.history = []

# Usage
history = ConversationHistory(max_length=10)
history.add("Hola", "Hola Rosario, ¿cómo está?")
history.add("Abre Netflix", "Abriendo Netflix para ti.")

formatted = history.get_formatted()
print(f"History length: {len(formatted)} messages")
```

```python Advanced History
import json
from datetime import datetime

class ConversationManager:
    def __init__(self, max_history=10, save_file=None):
        self.max_history = max_history
        self.save_file = save_file
        self.history = []
        
        if save_file and os.path.exists(save_file):
            self.load()
    
    def add(self, user_msg, ai_msg, metadata=None):
        """Add with metadata."""
        entry = {
            'timestamp': datetime.now().isoformat(),
            'user': user_msg,
            'assistant': ai_msg,
            'metadata': metadata or {}
        }
        self.history.append(entry)
        
        # Trim
        if len(self.history) > self.max_history:
            self.history.pop(0)
        
        # Auto-save
        if self.save_file:
            self.save()
    
    def save(self):
        """Save to file."""
        with open(self.save_file, 'w', encoding='utf-8') as f:
            json.dump(self.history, f, indent=2, ensure_ascii=False)
    
    def load(self):
        """Load from file."""
        with open(self.save_file, 'r', encoding='utf-8') as f:
            self.history = json.load(f)
    
    def get_recent(self, n=5):
        """Get last N entries."""
        return self.history[-n:]

# Usage
manager = ConversationManager(
    max_history=20,
    save_file='conversation_history.json'
)

manager.add(
    "Abre Netflix",
    "Abriendo Netflix",
    metadata={'app_opened': 'netflix', 'success': True}
)
```

    ```python
                    devices.append(device)
                    
        except socket.timeout:
            print("Network timeout during scan")
        except socket.error as e:
            print(f"Socket error: {e}")
        except Exception as e:
            print(f"Scan error: {e}")
        
        return devices
    ```
