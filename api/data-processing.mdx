## MFCC Feature Extraction

Mel-Frequency Cepstral Coefficients (MFCCs) are the primary audio features used for machine learning.

### extract_features()

Extracts MFCC features with automatic length normalization.

<ParamField path="audio_data" type="numpy.ndarray" required>
  Raw audio signal as 1D numpy array
</ParamField>

<ResponseField name="mfcc" type="numpy.ndarray">
  MFCC feature matrix with shape `(time_steps, n_mfcc)`
  - Automatically normalized to fixed length
  - Transposed for LSTM input format
</ResponseField>

<CodeGroup>
```python Basic Usage
import librosa
import numpy as np

def extract_features(audio_data, sample_rate=16000, duration=5, n_mfcc=13):
    """Extract MFCC features from audio."""
    # Normalize audio length
    target_len = int(sample_rate * duration)
    
    if len(audio_data) < target_len:
        # Pad with zeros if too short
        audio_data = np.pad(audio_data, (0, target_len - len(audio_data)))
    else:
        # Truncate if too long
        audio_data = audio_data[:target_len]
    
    # Extract MFCCs
    mfcc = librosa.feature.mfcc(
        y=audio_data,
        sr=sample_rate,
        n_mfcc=n_mfcc
    )
    
    return mfcc.T  # Transpose to (time_steps, n_mfcc)

# Example
audio, sr = librosa.load('sample.wav', sr=16000)
features = extract_features(audio)
print(f"Feature shape: {features.shape}")  # (time_steps, 13)
```

```python Advanced Configuration
import librosa

def extract_mfcc_advanced(
    audio_data,
    sample_rate=16000,
    n_mfcc=13,
    n_fft=2048,        # FFT window size
    hop_length=512,    # Hop length between frames
    n_mels=128         # Number of Mel bands
):
    """Extract MFCCs with custom parameters."""
    mfcc = librosa.feature.mfcc(
        y=audio_data,
        sr=sample_rate,
        n_mfcc=n_mfcc,
        n_fft=n_fft,
        hop_length=hop_length,
        n_mels=n_mels
    )
    
    # Apply delta features (velocity)
    mfcc_delta = librosa.feature.delta(mfcc)
    
    # Apply delta-delta features (acceleration)
    mfcc_delta2 = librosa.feature.delta(mfcc, order=2)
    
    # Stack all features
    features = np.vstack([mfcc, mfcc_delta, mfcc_delta2])
    
    return features.T

# Example with delta features
audio, sr = librosa.load('sample.wav', sr=16000)
features = extract_mfcc_advanced(audio)
print(f"Feature shape: {features.shape}")  # (time_steps, 39)
```

## Dataset Preparation

### prepare_dataset()

Loads and prepares complete training dataset from audio files.

<ResponseField name="return" type="tuple">
  Returns `(X: np.ndarray, y: np.ndarray)`
  - `X`: Feature arrays, shape `(n_samples, time_steps, n_mfcc)`
  - `y`: Binary labels (1 for positive, 0 for negative)
</ResponseField>

<CodeGroup>
```python Implementation
import os
import numpy as np
import librosa

def prepare_dataset(
    path_positive='audios',
    path_negative='audios bad',
    sample_rate=16000,
    duration=5,
    n_mfcc=13
):
    """Prepare dataset from audio files."""
    X = []
    y = []
    
    # Process positive samples (wake word)
    for path, label in [(path_positive, 1), (path_negative, 0)]:
        print(f"Loading {'positive' if label == 1 else 'negative'} samples...")
        
        if not os.path.exists(path):
            print(f"Warning: {path} does not exist")
            continue
        
        for filename in os.listdir(path):
            if filename.endswith('.wav'):
                filepath = os.path.join(path, filename)
                
                try:
                    # Load audio
                    audio, _ = librosa.load(
                        filepath,
                        sr=sample_rate,
                        duration=duration
                    )
                    
                    # Extract features
                    features = extract_features(audio)
                    
                    X.append(features)
                    y.append(label)
                    
                except Exception as e:
                    print(f"Error processing {filename}: {e}")
    
    return np.array(X), np.array(y)

# Usage
X_train, y_train = prepare_dataset()
print(f"Dataset shape: {X_train.shape}")
print(f"Labels: {np.unique(y_train, return_counts=True)}")
```

```python With Data Augmentation
import numpy as np
import librosa

def augment_audio(audio, sample_rate):
    """Apply data augmentation techniques."""
    augmented = []
    
    # Original
    augmented.append(audio)
    
    # Time stretch
    stretched = librosa.effects.time_stretch(audio, rate=1.1)
    augmented.append(stretched)
    
    # Pitch shift
    pitched = librosa.effects.pitch_shift(
        audio,
        sr=sample_rate,
        n_steps=2
    )
    augmented.append(pitched)
    
    # Add noise
    noise = np.random.randn(len(audio)) * 0.005
    noisy = audio + noise
    augmented.append(noisy)
    
    return augmented

def prepare_augmented_dataset():
    """Prepare dataset with augmentation."""
    X, y = [], []
    
    for path, label in [('audios', 1), ('audios bad', 0)]:
        for filename in os.listdir(path):
            if filename.endswith('.wav'):
                filepath = os.path.join(path, filename)
                audio, sr = librosa.load(filepath, sr=16000)
                
                # Generate augmented versions
                augmented_audios = augment_audio(audio, sr)
                
                for aug_audio in augmented_audios:
                    features = extract_features(aug_audio)
                    X.append(features)
                    y.append(label)
    
    return np.array(X), np.array(y)
```

```python Advanced Configuration
from sklearn.feature_extraction.text import TfidfVectorizer
import re

def spanish_tokenizer(text):
    """Custom tokenizer for Spanish text."""
    text = text.lower()
    text = re.sub(r'[^\w\sáéíóúñü]', '', text)
    tokens = text.split()
    return tokens

vectorizer = TfidfVectorizer(
    tokenizer=spanish_tokenizer,
    ngram_range=(1, 3),        # Unigrams, bigrams, trigrams
    max_features=5000,         # Limit vocabulary size
    min_df=2,                  # Ignore rare terms
    max_df=0.8,                # Ignore common terms

    use_idf=True,              # Enable IDF weighting

)
```

```python Feature Importance
from sklearn.feature_extraction.text import TfidfVectorizer
import numpy as np

def get_top_features(vectorizer, text, n=10):
    """Get top N features for given text."""
    # Transform text
    features = vectorizer.transform([text])
    
    # Get feature names and scores
    feature_names = vectorizer.get_feature_names_out()
    scores = features.toarray()[0]
    
    # Get top indices
    top_indices = np.argsort(scores)[::-1][:n]
    
    # Return top features
    return [
        (feature_names[i], scores[i])
        for i in top_indices
        if scores[i] > 0
    ]

# Example
vectorizer = TfidfVectorizer(ngram_range=(1, 2))
texts = ["abre netflix", "busca videos", "apaga tv"]
vectorizer.fit(texts)

test_text = "abre netflix por favor"
top_features = get_top_features(vectorizer, test_text)

print(f"Top features for: '{test_text}'")
for feature, score in top_features:
    print(f"  {feature}: {score:.4f}")
```

```python With Validation
import json

def load_and_validate_dataset(filepath):
    """Load dataset with validation."""
    with open(filepath, 'r', encoding='utf-8') as f:
        data = json.load(f)
    
    # Validate entries
    valid_data = []
    for i, item in enumerate(data):
        if 'text' in item and 'intent' in item:
            valid_data.append(item)
        else:
            print(f"Warning: Item {i} missing 'text' or 'intent'")

    # Extract texts and intents
    texts = [item['text'] for item in valid_data]
    intents = [item['intent'] for item in valid_data]
    
    return texts, intents, valid_data
```

```python Audio Text Cleaning
import re

def limpiar_texto_para_audio(texto):
    """Clean text for text-to-speech."""
    # Remove command tags
    texto = re.sub(r'/\*.*?\*/', '', texto)
    
    # Remove markdown formatting
    texto = texto.replace('*', '').replace('#', '').replace('_', ' ')
    
    # Remove URLs
    texto = re.sub(r'http\S+', '', texto)
    
    # Normalize whitespace
    texto = " ".join(texto.split())
    
    return texto

# Example
raw_text = "/*app(netflix)*/ Abriendo Netflix para ti, Rosario. https://netflix.com"
cleaned = limpiar_texto_para_audio(raw_text)
print(cleaned)  # "Abriendo Netflix para ti, Rosario."
```

```python Intent Text Preprocessing
import re
import unicodedata

def preprocess_intent_text(text):
    """Preprocess text for intent classification."""
    # Convert to lowercase
    text = text.lower()
    
    # Remove accents (optional)
    text = unicodedata.normalize('NFKD', text)
    text = ''.join([c for c in text if not unicodedata.combining(c)])
    
    # Remove punctuation except spaces
    text = re.sub(r'[^\w\s]', '', text)
    
    # Normalize whitespace
    text = ' '.join(text.split())
    
    return text

# Example
raw_text = "¡Abre Netflix, por favor!"
processed = preprocess_intent_text(raw_text)
print(processed)  # "abre netflix por favor"
```

### load_specific_datasets()

Load and normalize multiple dataset files.

```python Usage Example
normalizer = DataNormalizer()

# Load specific files
file_paths = [
    "datasets/conversations.json",
    "datasets/qa_pairs.json"
]

normalized_text = normalizer.load_specific_datasets(file_paths)
print(f"Loaded {len(normalized_text.split(normalizer.eos_token))} examples")
```

<ParamField path="file_paths" type="list[str]">
  List of JSON file paths to process
</ParamField>

**Returns**: Combined normalized text with entries separated by `\n\n`

<Note>
The normalizer handles both single JSON objects and arrays, as well as JSONL format (one JSON object per line).
</Note>
